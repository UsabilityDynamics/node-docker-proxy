function beforeValidate( data, done ) {
  module.debug( 'afterCreate' );

  // Convert names into a single string.
  if( data.RepoTags && data.RepoTags.length > 0 && !data.Name  ) {
    data.Name = data.RepoTags[0];
  }

  delete data.RepoTags;

  done();

}
